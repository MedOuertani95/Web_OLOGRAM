1. Check the status of srcmstr.service and ctrmc.service
# systemctl status ctrmc.service
# systemctl status srcmstr.service

2. Stop the services.
# systemctl stop ctrmc.service
# systemctl stop srcmstr.service

3. Disable autostart.
# systemctl disable ctrmc.service
# systemctl disable srcmstr.service

4. Check if autostart is disabled.
# systemctl is-enabled ctrmc.service
# systemctl is-enabled srcmstr.service
