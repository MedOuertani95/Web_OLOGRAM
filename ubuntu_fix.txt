# Solution 0 :
## launch terminal 
ctrl + alt + F2	
## enter superuser mode
sudo su
## configure all the Ubuntu packages that have been unpacked but not configured
dpkg --configure -a
## reinstall packages previously configured
apt-get -f install
## update
apt-get update
apt-get dist-upgrade
## reinstall the graphical environment
apt-get install --reinstall ubuntu-desktop
## delete unused packages and downloaded files
apt-get autoremove
apt-get clean
##  reboot the system
reboot
